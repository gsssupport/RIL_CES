CREATE OR REPLACE FUNCTION FN_TRG_HST_TESTINGFORM()
 RETURNS trigger AS 
 $BODY$ 
 DECLARE
 vSEQ BIGINT; 
 BEGIN 
 SELECT COALESCE(MAX(HST_ID),0)+1 INTO vSEQ FROM HST_TESTINGFORM ;
 IF TG_OP = 'INSERT' THEN 
 INSERT INTO HST_TESTINGFORM SELECT vSEQ , NEW.* ; 
 ELSIF TG_OP = 'UPDATE' THEN 
 INSERT INTO HST_TESTINGFORM SELECT vSEQ , NEW.* ; 

 END IF; 
 RETURN NEW; 
 END; 
 $BODY$ 
 LANGUAGE plpgsql ; 

 CREATE TRIGGER TRG_HST_TESTINGFORM
 AFTER INSERT Or UPDATE 
 ON TESTINGFORM
 FOR EACH ROW 
 EXECUTE PROCEDURE FN_TRG_HST_TESTINGFORM() ;


ALTER TABLE TESTINGFORM  DISABLE TRIGGER TRG_HST_TESTINGFORM ;
